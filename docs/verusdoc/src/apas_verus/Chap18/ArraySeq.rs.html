28    /// Data Type 18.1: Generic sequence trait for array-backed sequences.
29    pub trait ArraySeqTrait<T>: Sized {
30        /// - Create a new sequence of length `length` with each element initialized to `init_value`.
31        /// - Work Θ(length), Span Θ(1).
32        fn new(length: usize, init_value: T) -> Self where T: Clone;
33
34        /// - Set the element at `index` to `item` in place.
35        /// - Work Θ(1), Span Θ(1).
36        fn set(&mut self, index: usize, item: T) -> Result<(), &'static str>;
37
38        /// - Definition 18.1 (length). Return the number of elements.
39        /// - Work Θ(1), Span Θ(1).
40        fn length(&self) -> usize;
41
42        /// - Algorithm 19.11 (Function nth). Return a reference to the element at `index`.
43        /// - Work Θ(1), Span Θ(1).
44        fn nth(&self, index: usize) -> &T;
45
46        /// - Definition 18.1 (empty). Construct the empty sequence.
47        /// - Work Θ(1), Span Θ(1).
48        fn empty() -> Self;
49
50        /// - Definition 18.1 (singleton). Construct a singleton sequence containing `item`.
51        /// - Work Θ(1), Span Θ(1).
52        fn singleton(item: T) -> Self;
53
54        /// - Algorithm 18.3 (tabulate). Build a sequence by applying `f` to each index.
55        /// - Work Θ(length), Span Θ(1).
56        fn tabulate<F: Fn(usize) -> T>(f: &F, length: usize) -> ArraySeqS<T>;
57
58        /// - Algorithm 18.4 (map). Transform each element via `f`.
59        /// - Work Θ(|a|), Span Θ(1).
60        fn map<U: Clone, F: Fn(&T) -> U>(a: &ArraySeqS<T>, f: &F) -> ArraySeqS<U>;
61
62        /// - Definition 18.12 (subseq). Extract a contiguous subsequence.
63        /// - Work Θ(length), Span Θ(1).
64        fn subseq(a: &ArraySeqS<T>, start: usize, length: usize) -> Self where T: Clone;
65
66        /// - Definition 18.13 (append). Concatenate two sequences.
67        /// - Work Θ(|a| + |b|), Span Θ(1).
68        fn append(a: &ArraySeqS<T>, b: &ArraySeqS<T>) -> Self where T: Clone;
69
70        /// - Definition 18.14 (filter). Keep elements satisfying `pred`.
71        /// - Work Θ(|a|), Span Θ(1).
72        fn filter<F: Fn(&T) -> bool>(a: &ArraySeqS<T>, pred: &F) -> Self where T: Clone;
73
74        /// - Definition 18.15 (flatten). Concatenate a sequence of sequences.
75        /// - Work Θ(total length), Span Θ(1).
76        fn flatten(a: &ArraySeqS<ArraySeqS<T>>) -> Self where T: Clone;
77
78        /// - Definition 18.16 (update). Return a copy with the index replaced by the new value.
79        /// - Work Θ(|a|), Span Θ(1).
80        fn update(a: &ArraySeqS<T>, index: usize, item: T) -> Self where T: Clone;
81
82        /// - Definition 18.5 (isEmpty). true iff the sequence has length zero.
83        /// - Work Θ(1), Span Θ(1).
84        fn is_empty(&self) -> bool;
85
86        /// - Definition 18.5 (isSingleton). true iff the sequence has length one.
87        /// - Work Θ(1), Span Θ(1).
88        fn is_singleton(&self) -> bool;
89
90        /// - Definition 18.7 (iterate). Fold with accumulator `seed`.
91        /// - Work Θ(|a|), Span Θ(1).
92        fn iterate<A, F: Fn(&A, &T) -> A>(a: &ArraySeqS<T>, f: &F, seed: A) -> A;
93
94        /// - Definition 18.18 (reduce). Combine elements using associative `f` and identity `id`.
95        /// - Work Θ(|a|), Span Θ(1).
96        fn reduce<F: Fn(&T, &T) -> T>(a: &ArraySeqS<T>, f: &F, id: T) -> T where T: Clone;
97
98        /// - Definition 18.19 (scan). Prefix-reduce returning partial sums and total.
99        /// - Work Θ(|a|), Span Θ(1).
100        fn scan<F: Fn(&T, &T) -> T>(a: &ArraySeqS<T>, f: &F, id: T) -> (ArraySeqS<T>, T) where T: Clone;
101
102        /// - Definition 18.12 (subseq copy). Extract contiguous subsequence with allocation.
103        /// - Work Θ(length), Span Θ(1).
104        fn subseq_copy(&self, start: usize, length: usize) -> ArraySeqS<T> where T: Clone;
105
106        /// - Create sequence from Vec.
107        /// - Work Θ(n) worst case, Θ(1) best case, Span Θ(1).
108        fn from_vec(elts: Vec<T>) -> Self;
109    }

125    impl<T> View for ArraySeqIter<T> {
126        type V = (int, Seq<T>);
127        open spec fn view(&self) -> (int, Seq<T>) { (self.pos as int, self.elements@) }
128    }
129
130    pub open spec fn iter_invariant<T>(it: &ArraySeqIter<T>) -> bool { it.pos <= it.elements@.len() }
131
132    // See experiments/simple_seq_iter.rs::assumption_free_next for a version that proves
133    // without assume() by requiring iter_invariant. We can't add requires to Iterator::next in Verus.
134    // and Rust iterators have 70 functions on them making this sensible requirement impossible.
135    impl<T: Clone> Iterator for ArraySeqIter<T> {
136        type Item = T;
137
138        fn next(&mut self) -> (result: Option<T>)
139            ensures
140                self.pos <= self.elements.len(),
141                ({
142                    let (old_index, old_seq) = old(self)@;
143                    match result {
144                        None => {
145                            &&& self@ == old(self)@
146                            &&& old_index == old_seq.len()
147                            &&& self.pos == old_seq.len()
148                        },
149                        Some(element) => {
150                            let (new_index, new_seq) = self@;
151                            &&& 0 <= old_index < old_seq.len()
152                            &&& new_seq == old_seq
153                            &&& new_index == old_index + 1
154                            &&& vstd::pervasive::cloned(old_seq[old_index as int], element)
155                        },
156                    }
157                }),
158        {
159            if self.pos < self.elements.len() {
160                let elem = self.elements[self.pos].clone();
161                self.pos = self.pos + 1;
162                Some(elem)
163            } else {
164                assume(self.pos <= self.elements.len());
165                None
166            }
167        }
168    }

170    impl<T: View> ArraySeqS<T> {
171        pub fn new(length: usize, init_value: T) -> (result: ArraySeqS<T>)
172            where T: Clone
173            requires length <= usize::MAX
174            ensures result@.len() == length
175        {
176            let seq = vec![init_value; length];
177            ArraySeqS { seq }
178        }
179
180        pub fn set(&mut self, index: usize, item: T) -> (result: Result<(), &'static str>)
181            requires index < old(self).seq@.len()
182            ensures result.is_ok() ==> self.seq@.len() == old(self).seq@.len()
183        {
184            if index < self.seq.len() {
185                self.seq.set(index, item);
186                Ok(())
187            } else {
188                Err("Index out of bounds")
189            }
190        }
191
192        pub fn length(&self) -> (len: usize)
193            ensures len == self.seq@.len()
194        {
195            self.seq.len()
196        }
197
198        pub fn nth(&self, index: usize) -> (result: &T)
199            requires index < self.seq@.len()
200        {
201            &self.seq[index]
202        }
203
204        pub fn empty() -> (result: ArraySeqS<T>)
205            ensures result.seq@.len() == 0
206        {
207            ArraySeqS { seq: Vec::new() }
208        }
209
210        pub fn singleton(item: T) -> (result: ArraySeqS<T>)
211            ensures result.seq@.len() == 1
212        {
213            let mut seq = Vec::with_capacity(1);
214            seq.push(item);
215            ArraySeqS { seq }
216        }
217
218        pub fn tabulate<F: Fn(usize) -> T>(f: &F, length: usize) -> (result: ArraySeqS<T>)
219            requires 
220                length <= usize::MAX,
221                forall|i: usize| i < length ==> #[trigger] f.requires((i,)),
222            ensures result.seq@.len() == length
223        {
224            let mut seq = Vec::with_capacity(length);
225            let mut i: usize = 0;
226            while i < length
227                invariant
228                    i <= length,
229                    seq@.len() == i as int,
230                    forall|j: usize| j < length ==> #[trigger] f.requires((j,)),
231                decreases length - i,
232            {
233                seq.push(f(i));
234                i += 1;
235            }
236            ArraySeqS { seq }
237        }
238
239        pub fn map<U: Clone + View, F: Fn(&T) -> U>(a: &ArraySeqS<T>, f: &F) -> (result: ArraySeqS<U>)
240            requires forall|i: int| 0 <= i < a.seq@.len() ==> #[trigger] f.requires((&a.seq@[i],)),
241            ensures result.seq@.len() == a.seq@.len()
242        {
243            let len = a.seq.len();
244            let mut seq: Vec<U> = Vec::with_capacity(len);
245            let mut i: usize = 0;
246            while i < len
247                invariant
248                    i <= len,
249                    len == a.seq@.len(),
250                    seq@.len() == i as int,
251                    forall|j: int| 0 <= j < a.seq@.len() ==> #[trigger] f.requires((&a.seq@[j],)),
252                decreases len - i,
253            {
254                seq.push(f(&a.seq[i]));
255                i += 1;
256            }
257            ArraySeqS { seq }
258        }
259
260        pub fn isEmpty(&self) -> (empty: bool)
261            ensures empty <==> self.seq@.len() == 0
262        {
263            self.seq.len() == 0
264        }
265
266        pub fn isSingleton(&self) -> (single: bool)
267            ensures single <==> self.seq@.len() == 1
268        {
269            self.seq.len() == 1
270        }
271
272        pub fn from_vec(elts: Vec<T>) -> (result: ArraySeqS<T>)
273            ensures result.seq@ == elts@
274        {
275            ArraySeqS { seq: elts }
276        }
277
278        pub fn iter(&self) -> (it: ArraySeqIter<T>)
279            where T: Clone
280            ensures
281                it.elements@.len() == self.seq@.len(),
282                forall|i: int| 0 <= i < self.seq@.len() ==> cloned(self.seq@[i], #[trigger] it.elements@[i]),
283                it.pos == 0,
284                iter_invariant(&it),
285        {
286            ArraySeqIter { elements: self.seq.clone(), pos: 0 }
287        }
288    }

330        fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult {

332            for (i, item) in self.seq.iter().enumerate() {
333                if i > 0 { write!(f, ", ")?; }
