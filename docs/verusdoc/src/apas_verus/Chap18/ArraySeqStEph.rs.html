24    /// Base trait for single-threaded ephemeral array sequences (Chapter 18).
25    /// These methods are never redefined in later chapters.
26    pub trait ArraySeqStEphBaseTrait<T>: Sized {
27        /// Work Θ(n), Span Θ(1)
28        fn new(length: usize, init_value: T) -> Self where T: Clone;
29        /// Work Θ(1), Span Θ(1)
30        fn set(&mut self, index: usize, item: T) -> Result<(), &'static str>;
31        /// Work Θ(1), Span Θ(1)
32        fn length(&self) -> usize;
33        /// Work Θ(1), Span Θ(1)
34        fn nth(&self, index: usize) -> &T;
35        /// Work Θ(len), Span Θ(1)
36        fn subseq_copy(&self, start: usize, length: usize) -> Self where T: Clone;
37        /// Work Θ(Σ|a[i]|), Span Θ(1)
38        fn flatten(a: &ArraySeqStEphS<ArraySeqStEphS<T>>) -> Self where T: Clone;
39        /// Work Θ(n), Span Θ(1)
40        fn from_vec(elts: Vec<T>) -> Self;
41    }

90    pub open spec fn iter_invariant<T>(it: &ArraySeqStEphIter<T>) -> bool { it.pos <= it.elements@.len() }

92    // See experiments/simple_seq_iter.rs::assumption_free_next for a version that proves
93    // without assume() by requiring iter_invariant. We can't add requires to Iterator::next in Verus.
94    // and Rust iterators have 70 functions on them making this sensible requirement impossible.
95    impl<T: Clone> Iterator for ArraySeqStEphIter<T> {
96        type Item = T;
97
98        fn next(&mut self) -> (result: Option<T>)
99            ensures
100                self.pos <= self.elements.len(),
101                ({
102                    let (old_index, old_seq) = old(self)@;
103                    match result {
104                        None => {
105                            &&& self@ == old(self)@
106                            &&& old_index == old_seq.len()
107                            &&& self.pos == old_seq.len()
108                        },
109                        Some(element) => {
110                            let (new_index, new_seq) = self@;
111                            &&& 0 <= old_index < old_seq.len()
112                            &&& new_seq == old_seq
113                            &&& new_index == old_index + 1
114                            &&& vstd::pervasive::cloned(old_seq[old_index as int], element)
115                        },
116                    }
117                }),
118        {
119            if self.pos < self.elements.len() {
120                let elem = self.elements[self.pos].clone();
121                self.pos = self.pos + 1;
122                Some(elem)
123            } else {
124                assume(self.pos <= self.elements.len());
125                None
126            }
127        }
128    }

130    impl<T> ArraySeqStEphS<T> {
131        pub fn new(length: usize, init_value: T) -> (result: ArraySeqStEphS<T>)
132            where T: Clone
133            requires length <= usize::MAX
134            ensures result.seq@.len() == length
135        {
136            ArraySeqStEphS { seq: vec![init_value; length] }
137        }
138
139        pub fn set(&mut self, index: usize, item: T) -> (result: Result<(), &'static str>)
140            requires index < old(self).seq@.len()
141            ensures result.is_ok() ==> self.seq@.len() == old(self).seq@.len()
142        {
143            if index < self.seq.len() {
144                self.seq.set(index, item);
145                Ok(())
146            } else {
147                Err("Index out of bounds")
148            }
149        }
150
151        pub fn length(&self) -> (len: usize)
152            ensures len == self.seq@.len()
153        {
154            self.seq.len()
155        }
156
157        pub fn nth(&self, index: usize) -> (result: &T)
158            requires index < self.seq@.len()
159        {
160            &self.seq[index]
161        }
162
163        pub fn empty() -> (result: ArraySeqStEphS<T>)
164            ensures result.seq@.len() == 0
165        {
166            ArraySeqStEphS { seq: Vec::new() }
167        }
168
169        pub fn singleton(item: T) -> (result: ArraySeqStEphS<T>)
170            ensures result.seq@.len() == 1
171        {
172            let mut seq = Vec::with_capacity(1);
173            seq.push(item);
174            ArraySeqStEphS { seq }
175        }
176
177        pub fn tabulate<F: Fn(usize) -> T>(f: &F, length: usize) -> (result: ArraySeqStEphS<T>)
178            requires 
179                length <= usize::MAX,
180                forall|i: usize| i < length ==> #[trigger] f.requires((i,)),
181            ensures result.seq@.len() == length
182        {
183            let mut seq = Vec::with_capacity(length);
184            let mut i: usize = 0;
185            while i < length
186                invariant
187                    i <= length,
188                    seq@.len() == i as int,
189                    forall|j: usize| j < length ==> #[trigger] f.requires((j,)),
190                decreases length - i,
191            {
192                seq.push(f(i));
193                i += 1;
194            }
195            ArraySeqStEphS { seq }
196        }
197
198        pub fn map<U: Clone + View, F: Fn(&T) -> U>(a: &ArraySeqStEphS<T>, f: &F) -> (result: ArraySeqStEphS<U>)
199            requires forall|i: int| 0 <= i < a.seq@.len() ==> #[trigger] f.requires((&a.seq@[i],)),
200            ensures result.seq@.len() == a.seq@.len()
201        {
202            let len = a.seq.len();
203            let mut seq: Vec<U> = Vec::with_capacity(len);
204            let mut i: usize = 0;
205            while i < len
206                invariant
207                    i <= len,
208                    len == a.seq@.len(),
209                    seq@.len() == i as int,
210                    forall|j: int| 0 <= j < a.seq@.len() ==> #[trigger] f.requires((&a.seq@[j],)),
211                decreases len - i,
212            {
213                seq.push(f(&a.seq[i]));
214                i += 1;
215            }
216            ArraySeqStEphS { seq }
217        }
218
219        pub fn append(&self, b: &ArraySeqStEphS<T>) -> (result: ArraySeqStEphS<T>)
220            where T: Clone
221            requires self.seq@.len() + b.seq@.len() <= usize::MAX
222            ensures result.seq@.len() == self.seq@.len() + b.seq@.len()
223        {
224            let a_len = self.seq.len();
225            let b_len = b.seq.len();
226            let mut seq: Vec<T> = Vec::with_capacity(a_len + b_len);
227            let mut i: usize = 0;
228            while i < a_len
229                invariant
230                    i <= a_len,
231                    a_len == self.seq@.len(),
232                    seq@.len() == i as int,
233                decreases a_len - i,
234            {
235                seq.push(self.seq[i].clone());
236                i += 1;
237            }
238            let mut j: usize = 0;
239            while j < b_len
240                invariant
241                    j <= b_len,
242                    b_len == b.seq@.len(),
243                    seq@.len() == a_len + j,
244                decreases b_len - j,
245            {
246                seq.push(b.seq[j].clone());
247                j += 1;
248            }
249            ArraySeqStEphS { seq }
250        }
251
252        pub fn filter<F: Fn(&T) -> bool>(a: &ArraySeqStEphS<T>, pred: &F) -> (result: ArraySeqStEphS<T>)
253            where T: Clone
254            requires forall|i: int| 0 <= i < a.seq@.len() ==> #[trigger] pred.requires((&a.seq@[i],)),
255            ensures result.seq@.len() <= a.seq@.len()
256        {
257            let len = a.seq.len();
258            let mut seq: Vec<T> = Vec::new();
259            let mut i: usize = 0;
260            while i < len
261                invariant
262                    i <= len,
263                    len == a.seq@.len(),
264                    seq@.len() <= i,
265                    forall|j: int| 0 <= j < a.seq@.len() ==> #[trigger] pred.requires((&a.seq@[j],)),
266                decreases len - i,
267            {
268                if pred(&a.seq[i]) {
269                    seq.push(a.seq[i].clone());
270                }
271                i += 1;
272            }
273            ArraySeqStEphS { seq }
274        }
275
276        pub fn isEmpty(&self) -> (empty: bool)
277            ensures empty <==> self.seq@.len() == 0
278        {
279            self.seq.len() == 0
280        }
281
282        pub fn isSingleton(&self) -> (single: bool)
283            ensures single <==> self.seq@.len() == 1
284        {
285            self.seq.len() == 1
286        }
287
288        pub fn from_vec(elts: Vec<T>) -> (result: ArraySeqStEphS<T>)
289            ensures result.seq@ == elts@
290        {
291            ArraySeqStEphS { seq: elts }
292        }
293
294        pub fn iter(&self) -> (it: ArraySeqStEphIter<T>)
295            where T: Clone
296            ensures
297                it.elements@.len() == self.seq@.len(),
298                forall|i: int| 0 <= i < self.seq@.len() ==> cloned(self.seq@[i], #[trigger] it.elements@[i]),
299                it.pos == 0,
300                iter_invariant(&it),
301        {
302            ArraySeqStEphIter { elements: self.seq.clone(), pos: 0 }
303        }
304
305        pub fn subseq(&self, start: usize, length: usize) -> (result: ArraySeqStEphS<T>)
306            where T: Clone
307            requires 
308                start + length <= self.seq@.len(),
309                self.seq@.len() <= usize::MAX as int,
310            ensures result.seq@.len() == length
311        {
312            let end = start + length;
313            let mut seq: Vec<T> = Vec::with_capacity(length);
314            let mut i: usize = start;
315            while i < end
316                invariant
317                    start <= i <= end,
318                    end == start + length,
319                    end <= self.seq@.len(),
320                    seq@.len() == (i - start) as int,
321                decreases end - i,
322            {
323                seq.push(self.seq[i].clone());
324                i += 1;
325            }
326            ArraySeqStEphS { seq }
327        }
328
329        pub fn reduce<F: Fn(&T, &T) -> T>(a: &ArraySeqStEphS<T>, f: &F, id: T) -> (result: T)
330            where T: Clone
331            requires forall|x: &T, y: &T| #[trigger] f.requires((x, y)),
332        {
333            let len = a.seq.len();
334            let mut acc = id;
335            let mut i: usize = 0;
336            while i < len
337                invariant
338                    i <= len,
339                    len == a.seq@.len(),
340                    forall|x: &T, y: &T| #[trigger] f.requires((x, y)),
341                decreases len - i,
342            {
343                acc = f(&acc, &a.seq[i]);
344                i += 1;
345            }
346            acc
347        }
348
349        pub fn iterate<A, F: Fn(&A, &T) -> A>(a: &ArraySeqStEphS<T>, f: &F, seed: A) -> (result: A)
350            requires forall|x: &A, y: &T| #[trigger] f.requires((x, y)),
351        {
352            let len = a.seq.len();
353            let mut acc = seed;
354            let mut i: usize = 0;
355            while i < len
356                invariant
357                    i <= len,
358                    len == a.seq@.len(),
359                    forall|x: &A, y: &T| #[trigger] f.requires((x, y)),
360                decreases len - i,
361            {
362                acc = f(&acc, &a.seq[i]);
363                i += 1;
364            }
365            acc
366        }
