36    // Chapter 19 trait - provides alternative algorithmic implementations
37    // Import and use this trait to get Chapter 19's algorithms
38    pub trait ArraySeqStPerTrait<T: View + Clone>: Sized {
39        spec fn spec_len(&self) -> nat;
40
41        fn empty() -> (result: Self)
42            ensures result.spec_len() == 0;
43
44        fn singleton(item: T) -> (result: Self)
45            ensures result.spec_len() == 1;
46
47        fn map<U: Clone + View, F: Fn(&T) -> U>(a: &Self, f: &F) -> ArraySeqStPerS<U>
48            requires
49                a.spec_len() <= usize::MAX as int,
50                forall|i: int| 0 <= i < a.spec_len() ==> #[trigger] f.requires((&a.nth_spec(i),));
51
52        fn append(a: &Self, b: &Self) -> (result: Self)
53            requires a.spec_len() + b.spec_len() <= usize::MAX as int
54            ensures result.spec_len() == a.spec_len() + b.spec_len();
55
56        fn filter<F: Fn(&T) -> bool>(a: &Self, pred: &F) -> (result: Self)
57            requires
58                a.spec_len() <= usize::MAX as int,
59                forall|i: int| 0 <= i < a.spec_len() ==> #[trigger] pred.requires((&a.nth_spec(i),))
60            ensures result.spec_len() <= a.spec_len();
61
62        fn update(a: &Self, index: usize, item: T) -> (result: Self)
63            requires
64                index < a.spec_len(),
65                a.spec_len() <= usize::MAX as int,
66            ensures result.spec_len() == a.spec_len();
67
68        fn is_empty(a: &Self) -> (empty: bool)
69            ensures empty <==> a.spec_len() == 0;
70
71        fn is_singleton(a: &Self) -> (single: bool)
72            ensures single <==> a.spec_len() == 1;
73
74        fn iterate<A, F: Fn(&A, &T) -> A>(a: &Self, f: &F, seed: A) -> A
75            requires forall|x: &A, y: &T| #[trigger] f.requires((x, y));
76
77        fn reduce<F: Fn(&T, &T) -> T>(a: &Self, f: &F, id: T) -> T
78            requires
79                a.spec_len() <= usize::MAX as int,
80                forall|x: &T, y: &T| #[trigger] f.requires((x, y));
81
82        fn scan<F: Fn(&T, &T) -> T>(a: &Self, f: &F, id: T) -> (result: (Self, T))
83            requires forall|x: &T, y: &T| #[trigger] f.requires((x, y))
84            ensures result.0.spec_len() == a.spec_len();
85
86        fn select<'a>(a: &'a Self, b: &'a Self, i: usize) -> (result: Option<&'a T>)
87            ensures
88                i < a.spec_len() ==> result.is_some(),
89                a.spec_len() <= i < a.spec_len() + b.spec_len() ==> result.is_some();
90
91        fn append_select(a: &Self, b: &Self) -> (result: Self)
92            requires a.spec_len() + b.spec_len() <= usize::MAX as int
93            ensures result.spec_len() == a.spec_len() + b.spec_len();
94
95        spec fn nth_spec(&self, i: int) -> T;
96
97        fn from_set(set: &SetStEph<T>) -> (result: Self)
98            where T: StT + Hash
99            requires valid_key_type::<T>()
100            ensures result.spec_len() == set@.len();
101    }

103    impl<T: View + Clone> ArraySeqStPerTrait<T> for ArraySeqStPerS<T> {
104
105        open spec fn spec_len(&self) -> nat {
106            self.seq@.len()
107        }
108
109        open spec fn nth_spec(&self, i: int) -> T {
110            self.seq@[i]
111        }
112
113        // Algorithm 19.1: empty = tabulate(lambda i._, 0)
114        fn empty() -> ArraySeqStPerS<T> {
115            ArraySeqStPerS::<T>::empty()
116        }
117
118        // Algorithm 19.2: singleton x = tabulate(lambda i.x, 1)
119        fn singleton(item: T) -> ArraySeqStPerS<T> {
120            ArraySeqStPerS::<T>::tabulate(&|_i: usize| item.clone(), 1)
121        }
122
123        // Algorithm 19.3: map f a = tabulate(lambda i.f(a[i]), |a|)
124        fn map<U: Clone + View, F: Fn(&T) -> U>(a: &ArraySeqStPerS<T>, f: &F) -> ArraySeqStPerS<U> {
125            let n = a.length();
126            proof {
127                // Connect outer requires to closure requires
128                assert forall|i: usize| i < n implies (i as int) < a.spec_len() && #[trigger] f.requires((&a.seq@[i as int],)) by {
129                    assert(a.nth_spec(i as int) == a.seq@[i as int]);
130                }
131            }
132            ArraySeqStPerS::<U>::tabulate(
133                &(|i: usize| -> (r: U)
134                    requires
135                        (i as int) < a.spec_len(),
136                        f.requires((&a.seq@[i as int],)),
137                {
138                    let elem = a.nth(i);
139                    f(elem)
140                }),
141                n,
142            )
143        }
144
145        // Algorithm 19.4: append a b = flatten([a, b])
146        fn append(a: &ArraySeqStPerS<T>, b: &ArraySeqStPerS<T>) -> ArraySeqStPerS<T> {
147            // Construct pair manually to avoid opaque f.ensures
148            let a_clone = a.clone_plus();
149            let b_clone = b.clone_plus();
150            proof {
151                assert(a_clone.seq@ =~= a.seq@);
152                assert(b_clone.seq@ =~= b.seq@);
153            }
154            let mut pair_vec: Vec<ArraySeqStPerS<T>> = Vec::with_capacity(2);
155            pair_vec.push(a_clone);
156            pair_vec.push(b_clone);
157            let pair = ArraySeqStPerS::<ArraySeqStPerS<T>> { seq: pair_vec };
158            proof {
159                assert(pair.spec_len() == 2);
160                assert(pair.seq@[0].seq@.len() == a.seq@.len());
161                assert(pair.seq@[1].seq@.len() == b.seq@.len());
162            }
163            flatten(&pair)
164        }
165
166        // Algorithm 19.5: filter f a = flatten(map(deflate f, a))
167        fn filter<F: Fn(&T) -> bool>(a: &ArraySeqStPerS<T>, pred: &F) -> ArraySeqStPerS<T> {
168            // Build deflated array manually to avoid opaque f.ensures
169            let n = a.length();
170            let mut deflated_vec: Vec<ArraySeqStPerS<T>> = Vec::with_capacity(n);
171            proof {
172                // Connect outer requires (nth_spec) to seq@
173                assert forall|j: usize| j < n implies #[trigger] pred.requires((&a.seq@[j as int],)) by {
174                    assert(a.nth_spec(j as int) == a.seq@[j as int]);
175                }
176            }
177            let mut i: usize = 0;
178            while i < n
179                invariant
180                    i <= n,
181                    n == a.seq@.len(),
182                    deflated_vec@.len() == i as int,
183                    forall|j: int| 0 <= j < i as int ==> #[trigger] deflated_vec@[j].seq@.len() <= 1,
184                    forall|j: usize| j < n ==> #[trigger] pred.requires((&a.seq@[j as int],)),
185                decreases n - i,
186            {
187                let elem = a.nth(i);
188                let deflated_elem = deflate(pred, elem);
189                proof {
190                    assert(deflated_elem.seq@.len() <= 1);
191                }
192                deflated_vec.push(deflated_elem);
193                i += 1;
194            }
195            let deflated = ArraySeqStPerS::<ArraySeqStPerS<T>> { seq: deflated_vec };
196            flatten(&deflated)
197        }
198
199        // Algorithm 19.6: update a (i, x) = tabulate(lambda j. if i = j then x else a[j], |a|)
200        fn update(a: &ArraySeqStPerS<T>, index: usize, item: T) -> ArraySeqStPerS<T> {
201            ArraySeqStPerS::<T>::tabulate(
202                &(|j: usize| -> (r: T)
203                    requires j < a.seq@.len()
204                {
205                    if j == index { item.clone() } else { a.nth(j).clone() }
206                }),
207                a.length(),
208            )
209        }
210
211        // Algorithm 19.7: is_empty a = |a| = 0
212        fn is_empty(a: &ArraySeqStPerS<T>) -> bool {
213            a.length() == 0
214        }
215
216        // Algorithm 19.7: is_singleton a = |a| = 1
217        fn is_singleton(a: &ArraySeqStPerS<T>) -> bool {
218            a.length() == 1
219        }
220
221        // Algorithm 19.8: iterate f x a = left-to-right traversal
222        fn iterate<A, F: Fn(&A, &T) -> A>(a: &ArraySeqStPerS<T>, f: &F, seed: A) -> A {
223            let len = a.seq.len();
224            let mut acc = seed;
225            let mut i: usize = 0;
226            while i < len
227                invariant
228                    i <= len,
229                    len == a.seq@.len(),
230                    forall|x: &A, y: &T| #[trigger] f.requires((x, y)),
231                decreases len - i,
232            {
233                acc = f(&acc, a.nth(i));
234                i += 1;
235            }
236            acc
237        }
238
239        // Algorithm 19.9: reduce using divide-and-conquer
240        fn reduce<F: Fn(&T, &T) -> T>(a: &ArraySeqStPerS<T>, f: &F, id: T) -> T
241            decreases a.seq@.len()
242        {
243            let len = a.length();
244            if len == 0 {
245                id
246            } else if len == 1 {
247                a.nth(0).clone()
248            } else {
249                let mid = len / 2;
250                let left = a.subseq_copy(0, mid);
251                let right = a.subseq_copy(mid, len - mid);
252                let left_result = <ArraySeqStPerS<T> as ArraySeqStPerTrait<T>>::reduce(&left, f, id.clone());
253                let right_result = <ArraySeqStPerS<T> as ArraySeqStPerTrait<T>>::reduce(&right, f, id);
254                f(&left_result, &right_result)
255            }
256        }
257
258        // Algorithm 19.10: scan using contraction (simplified sequential version)
259        fn scan<F: Fn(&T, &T) -> T>(a: &ArraySeqStPerS<T>, f: &F, id: T) -> (ArraySeqStPerS<T>, T) {
260            let len = a.seq.len();
261            let mut acc = id.clone();
262            let mut results: Vec<T> = Vec::with_capacity(len);
263            let mut i: usize = 0;
264            while i < len
265                invariant
266                    i <= len,
267                    len == a.seq@.len(),
268                    results@.len() == i as int,
269                    forall|x: &T, y: &T| #[trigger] f.requires((x, y)),
270                decreases len - i,
271            {
272                acc = f(&acc, a.nth(i));
273                results.push(acc.clone());
274                i += 1;
275            }
276            (ArraySeqStPerS { seq: results }, acc)
277        }
278
279        // select helper for append_select
280        fn select<'a>(a: &'a ArraySeqStPerS<T>, b: &'a ArraySeqStPerS<T>, i: usize) -> Option<&'a T> {
281            let len_a = a.length();
282            if i < len_a {
283                Some(a.nth(i))
284            } else {
285                let offset = i - len_a;
286                let len_b = b.length();
287                if offset < len_b {
288                    Some(b.nth(offset))
289                } else {
290                    None
291                }
292            }
293        }
294
295        // Algorithm 19.4 alternative: append a b = tabulate(select(a,b), |a|+|b|)
296        fn append_select(a: &ArraySeqStPerS<T>, b: &ArraySeqStPerS<T>) -> ArraySeqStPerS<T> {
297            let total = a.length() + b.length();
298            ArraySeqStPerS::<T>::tabulate(
299                &(|i: usize| -> (r: T)
300                    requires i < a.seq@.len() + b.seq@.len()
301                {
302                    let opt = <ArraySeqStPerS<T> as ArraySeqStPerTrait<T>>::select(a, b, i);
303                    proof { assert(opt.is_some()); }
304                    opt.unwrap().clone()
305                }),
306                total,
307            )
308        }
309
310        fn from_set(set: &SetStEph<T>) -> (result: ArraySeqStPerS<T>)
311            where T: StT + Hash
312        {
313            let seq = set.to_seq();
314            proof {
315                // to_seq ensures: seq@.no_duplicates() && bijection with set@
316                // Since no_duplicates and same elements, lengths must match
317                assert(seq@.no_duplicates());
318                assume(seq@.len() == set@.len()); // TODO: prove via bijection lemma
319            }
320            ArraySeqStPerS { seq }
321        }
322    }

324    // Helper: flatten - sums lengths of inner sequences
325    #[verifier::external_body]
326    fn flatten<T: View + Clone>(ss: &ArraySeqStPerS<ArraySeqStPerS<T>>) -> (result: ArraySeqStPerS<T>)
327        ensures
328            // For 2-element case (append)
329            ss.seq@.len() == 2 ==> result.seq@.len() == ss.seq@[0].seq@.len() + ss.seq@[1].seq@.len(),
330            // For filter case: result <= sum of inner lengths <= outer length (each inner <= 1)
331            (forall|i: int| #![auto] 0 <= i < ss.seq@.len() ==> ss.seq@[i].seq@.len() <= 1)
332                ==> result.seq@.len() <= ss.seq@.len(),
333    {
334        let mut total_len: usize = 0;
335        let ss_len = ss.seq.len();
336        for i in 0..ss_len {
337            total_len = total_len + ss.seq[i].seq.len();
338        }
339        let mut result: Vec<T> = Vec::with_capacity(total_len);
340        for j in 0..ss_len {
341            let inner = &ss.seq[j];
342            for k in 0..inner.seq.len() {
343                result.push(inner.seq[k].clone());
344            }
345        }
346        ArraySeqStPerS { seq: result }
347    }
348
349    // Helper: deflate for filter
350    fn deflate<T: View + Clone, F: Fn(&T) -> bool>(pred: &F, x: &T) -> (result: ArraySeqStPerS<T>)
351        requires pred.requires((x,))
352        ensures result.seq@.len() <= 1
353    {
354        if pred(x) {
355            ArraySeqStPerS::<T>::singleton(x.clone())
356        } else {
357            ArraySeqStPerS::<T>::empty()
358        }
359    }

377    pub trait ArraySeqStPerTrait<T: Clone> {
378        fn empty() -> Self;
379        fn singleton(item: T) -> Self;
380        fn map<U: Clone, F: Fn(&T) -> U>(a: &Self, f: &F) -> ArraySeqStPerS<U>;
381        fn append(a: &Self, b: &Self) -> Self;
382        fn filter<F: Fn(&T) -> bool>(a: &Self, pred: &F) -> Self;
383        fn update(a: &Self, index: usize, item: T) -> Self;
384        fn is_empty(a: &Self) -> bool;
385        fn is_singleton(a: &Self) -> bool;
386        fn iterate<A, F: Fn(&A, &T) -> A>(a: &Self, f: &F, seed: A) -> A;
387        fn reduce<F: Fn(&T, &T) -> T>(a: &Self, f: &F, id: T) -> T;
388        fn scan<F: Fn(&T, &T) -> T>(a: &Self, f: &F, id: T) -> (Self, T) where Self: Sized;
389        fn select<'a>(a: &'a Self, b: &'a Self, i: usize) -> Option<&'a T>;
390        fn append_select(a: &Self, b: &Self) -> Self;
391        fn from_set(set: &SetStEph<T>) -> Self where T: StT + Hash;
392    }

394    #[cfg(not(verus_keep_ghost))]
395    impl<T: Clone> ArraySeqStPerTrait<T> for ArraySeqStPerS<T> {
396        fn empty() -> Self { Self::from_vec(Vec::new()) }
397        fn singleton(item: T) -> Self { Self::from_vec(vec![item]) }
398        fn map<U: Clone, F: Fn(&T) -> U>(a: &Self, f: &F) -> ArraySeqStPerS<U> {
399            ArraySeqStPerS::<U>::tabulate(&|i| f(a.nth(i)), a.length())
400        }
401        fn append(a: &Self, b: &Self) -> Self {
402            let mut seq = a.seq.clone();
403            seq.extend(b.seq.iter().cloned());
404            Self { seq }
405        }
406        fn filter<F: Fn(&T) -> bool>(a: &Self, pred: &F) -> Self {
407            Self { seq: a.seq.iter().filter(|x| pred(x)).cloned().collect() }
408        }
409        fn update(a: &Self, index: usize, item: T) -> Self {
410            Self::tabulate(&|j| if j == index { item.clone() } else { a.nth(j).clone() }, a.length())
411        }
412        fn is_empty(a: &Self) -> bool { a.length() == 0 }
413        fn is_singleton(a: &Self) -> bool { a.length() == 1 }
414        fn iterate<A, F: Fn(&A, &T) -> A>(a: &Self, f: &F, seed: A) -> A {
415            a.seq.iter().fold(seed, |acc, x| f(&acc, x))
416        }
417        fn reduce<F: Fn(&T, &T) -> T>(a: &Self, f: &F, id: T) -> T {
418            if a.length() == 0 { id }
419            else if a.length() == 1 { a.nth(0).clone() }
420            else {
421                let mid = a.length() / 2;
422                let left = a.subseq_copy(0, mid);
423                let right = a.subseq_copy(mid, a.length() - mid);
424                f(&Self::reduce(&left, f, id.clone()), &Self::reduce(&right, f, id))
425            }
426        }
427        fn scan<F: Fn(&T, &T) -> T>(a: &Self, f: &F, id: T) -> (Self, T) {
428            let mut acc = id;
429            let seq: Vec<T> = a.seq.iter().map(|x| { acc = f(&acc, x); acc.clone() }).collect();
430            (Self { seq }, acc)
431        }
432        fn select<'a>(a: &'a Self, b: &'a Self, i: usize) -> Option<&'a T> {
433            if i < a.length() { Some(a.nth(i)) }
434            else if i - a.length() < b.length() { Some(b.nth(i - a.length())) }
435            else { None }
436        }
437        fn append_select(a: &Self, b: &Self) -> Self {
438            Self::tabulate(&|i| Self::select(a, b, i).unwrap().clone(), a.length() + b.length())
439        }
440        fn from_set(set: &SetStEph<T>) -> Self where T: StT + Hash {
441            Self { seq: set.to_seq() }
442        }
443    }

445    #[cfg(not(verus_keep_ghost))]
446    fn flatten<T: Clone>(ss: &ArraySeqStPerS<ArraySeqStPerS<T>>) -> ArraySeqStPerS<T> {
447        let mut result = Vec::new();
448        for inner in ss.seq.iter() {
449            result.extend(inner.seq.iter().cloned());
450        }
451        ArraySeqStPerS { seq: result }
452    }
453
454    #[cfg(not(verus_keep_ghost))]
455    fn deflate<T: Clone, F: Fn(&T) -> bool>(pred: &F, x: &T) -> ArraySeqStPerS<T> {
456        if pred(x) { ArraySeqStPerS::singleton(x.clone()) } else { ArraySeqStPerS::empty() }
457    }
