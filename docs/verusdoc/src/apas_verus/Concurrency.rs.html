27    // StTInMtT: St-friendly elements that can be shared across threads (StT + Send + Sync)
28    pub trait StTInMtT: StT + Send + Sync {}
29
30    // MtT: multi-threaded friendly elements; minimal so it can include Mutex<..>
31    // Keep only thread-safety and size requirements.
32    //
33    // NOTE: clone_mt() has no Verus specification. For verified code, use clone_plus()
34    // from ClonePlus trait instead, which has `ensures cloned(*self, res)` postcondition.
35    // All types implementing MtT also implement Clone, so clone_plus() is always available.
36    pub trait MtT: Sized + Send + Sync {
37        type Inner: StT;
38        /// Deprecated for verified code - use clone_plus() which has specification.
39        fn clone_mt(&self) -> Self;
40        fn new_mt(inner: Self::Inner) -> Self;
41    }
42
43    // MtKey: Multi-threaded key type with ordering and static lifetime
44    pub trait MtKey: StTInMtT + Ord + 'static {}
45
46    // MtVal: Multi-threaded value type with static lifetime
47    pub trait MtVal: StTInMtT + 'static {}
48
49    // MtFn: Multi-threaded function type with common bounds
50    pub trait MtFn<Args, Output>: Fn(Args) -> Output + Send + Sync + 'static {}
51
52    // MtFnClone: Multi-threaded function type with Clone
53    pub trait MtFnClone<Args, Output>: Fn(Args) -> Output + Send + Sync + Clone + 'static {}
54
55    // MtReduceFn: Multi-threaded reducer function type
56    pub trait MtReduceFn<V>: Fn(&V, &V) -> V + Clone + Send + Sync + 'static {}
57
58    // PredMt: Multi-threaded predicate function (boolean function)
59    pub trait PredMt<T>: Fn(&T) -> B + Send + Sync + 'static {}
60
61    // PredVal: Multi-threaded predicate function taking values by value
62    pub trait PredVal<T>: Fn(T) -> B + Send + Sync + 'static {}
63
64    // Backward compatibility alias
65    pub use PredMt as Pred;
66
67    // Blanket impls
68    impl<T> StTInMtT for T where T: StT + Send + Sync {}
69    impl<T> MtKey for T where T: StTInMtT + Ord + 'static {}
70    impl<T> MtVal for T where T: StTInMtT + 'static {}
71    impl<T, Args, Output> MtFn<Args, Output> for T where T: Fn(Args) -> Output + Send + Sync + 'static {}
72    impl<T, Args, Output> MtFnClone<Args, Output> for T where T: Fn(Args) -> Output + Send + Sync + 'static + Clone {}
73    impl<T, V> MtReduceFn<V> for T where T: Fn(&V, &V) -> V + Clone + Send + Sync + 'static {}
74    impl<F, T> PredMt<T> for F where F: Fn(&T) -> B + Send + Sync + 'static {}
75    impl<F, T> PredVal<T> for F where F: Fn(T) -> B + Send + Sync + 'static {}

77    // MtT for Pair when components are Send+Sync+Clone
78    impl<A: StT + Send + Sync + Clone, B: StT + Send + Sync + Clone> MtT for Pair<A, B> {
79        type Inner = Pair<A, B>;
80        fn clone_mt(&self) -> Self { Pair(self.0.clone_plus(), self.1.clone_plus()) }
81        fn new_mt(inner: Self::Inner) -> Self { inner }
82    }

147    // MtT implementations for Mutex - outside verus! because Mutex is not specified
148    impl<T: StT + Send> MtT for Mutex<T> {
149        type Inner = T;
150        fn clone_mt(&self) -> Self {
151            let inner = self.lock().unwrap().clone();
152            Mutex::new(inner)
153        }
154        fn new_mt(inner: Self::Inner) -> Self { Mutex::new(inner) }
155    }
