2//! Extensions to vstd::arithmetic::power2 for powers of 2.
3
4#[cfg(verus_keep_ghost)]
5pub mod power2_plus {
6    use vstd::prelude::*;
7    use vstd::arithmetic::power2::{pow2, lemma_pow2_pos, lemma_pow2_strictly_increases, lemma2_to64_rest};
8
9    verus! {
10
11        /// pow2 is monotonic: if a <= b then pow2(a) <= pow2(b)
12// Veracity: USED
13        pub proof fn lemma_pow2_mono(a: nat, b: nat)
14            requires a <= b,
15            ensures pow2(a) <= pow2(b),
16        {
17            if a < b {
18                lemma_pow2_strictly_increases(a, b);
19            }
20        }
21
22        /// pow2(46) < u64::MAX
23        /// This is provable via lemma2_to64_rest which gives pow2(46) == 0x400000000000
24// Veracity: USED
25        pub proof fn lemma_pow2_46_lt_u64_max()
26            ensures pow2(46) < u64::MAX as nat,
27        {
28            lemma2_to64_rest();
29// Veracity: UNNEEDED assert             assert(pow2(46) == 0x400000000000);
30// Veracity: UNNEEDED assert             assert(u64::MAX == 0xFFFFFFFFFFFFFFFF);
31        }
32
33        /// pow2(63) < u64::MAX (since u64::MAX = 2^64 - 1)
34// Veracity: USED
35        pub proof fn lemma_pow2_63_lt_u64_max()
36            ensures pow2(63) < u64::MAX as nat,
37        {
38            lemma2_to64_rest();
39// Veracity: UNNEEDED assert             assert(pow2(63) == 0x8000000000000000);
40// Veracity: UNNEEDED assert             assert(u64::MAX == 0xFFFFFFFFFFFFFFFF);
41        }
42
43// Veracity: USED
44        /// pow2(n) < u64::MAX for n < 64
45        pub proof fn lemma_pow2_lt_u64_max(n: nat)
46            requires n < 64,
47            ensures pow2(n) < u64::MAX as nat,
48        {
49            lemma_pow2_63_lt_u64_max();
50            if n < 63 {
51                lemma_pow2_strictly_increases(n, 63);
52            }
53        }
54// Veracity: USED
55
56        /// pow2(31) < u32::MAX
57        pub proof fn lemma_pow2_31_lt_u32_max()
58            ensures pow2(31) < u32::MAX as nat,
59        {
60            vstd::arithmetic::power2::lemma2_to64();
61// Veracity: UNNEEDED assert             assert(pow2(31) == 0x80000000);
62// Veracity: UNNEEDED assert             assert(u32::MAX == 0xFFFFFFFF);
63        }
64
65        /// pow2(n) < u32::MAX for n < 32
66// Veracity: USED
67        pub proof fn lemma_pow2_lt_u32_max(n: nat)
68            requires n < 32,
69            ensures pow2(n) < u32::MAX as nat,
70        {
71            lemma_pow2_31_lt_u32_max();
72            if n < 31 {
73                lemma_pow2_strictly_increases(n, 31);
74            }
75        }
76
77    } // verus!
78} // mod
