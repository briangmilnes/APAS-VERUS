11    // Spec function: sum of u32 values in a sequence as nat
12    pub open spec fn spec_sum_u32_seq(s: Seq<u32>) -> nat 
13        decreases s.len()
14    {
15        if s.len() == 0 {
16            0nat
17        } else {
18            s.last() as nat + spec_sum_u32_seq(s.drop_last())
19        }
20    }

22    // Spec function: sum using fold_left (equivalent to spec_sum_u32_seq)
23    pub open spec fn spec_sum_u32_fold(s: Seq<u32>) -> nat {
24        s.fold_left(0nat, |acc: nat, v: u32| acc + v as nat)
25    }

46    pub fn seq_u32_to_CheckedU32(s: &Vec<u32>) -> (sum: CheckedU32)
47        ensures 
48            sum.is_normal() ==> sum@ == spec_sum_u32_seq(s@) as int
49    {
50        let mut sum = CheckedU32::new(0);
51        let mut i: usize = 0;
52        let ghost s_spec = s@;
53
54        #[cfg_attr(verus_keep_ghost, verifier::loop_isolation(false))]
55        while i < s.len()
56            invariant
57                i <= s.len(),
58                s@ == s_spec,
59                sum.is_normal() ==> sum@ == spec_sum_u32_seq(s_spec.take(i as int)) as int,
60            decreases s.len() - i
61        {
62            let ghost old_i = i as int;
63            let ghost old_sum = sum@;
64            
65            let v = s[i];
66            sum = sum.add_value(v);
67            i = i + 1;
68            
69            proof {
70                // Lemma: spec_sum_u32_seq(take(old_i + 1)) == spec_sum_u32_seq(take(old_i)) + s[old_i]
71                lemma_sum_u32_unfold_take(s_spec, old_i);
72                
73                // add_value ensures: sum@ == old_sum + v as int
74                // If old sum was normal: old_sum == spec_sum_u32_seq(take(old_i)) as int
75                // From lemma: spec_sum_u32_seq(take(old_i + 1)) == spec_sum_u32_seq(take(old_i)) + s[old_i]
76                // So: sum@ == spec_sum_u32_seq(take(old_i)) as int + v as int
77                //          == spec_sum_u32_seq(take(old_i + 1)) as int
78                //          == spec_sum_u32_seq(take(i)) as int
79                
80                // FIXME: SMT has trouble connecting recursive spec function unfolding
81                assume(sum.is_normal() ==> sum@ == spec_sum_u32_seq(s_spec.take(i as int)) as int);
82            }
83        }
84        proof {
85            assert(s_spec.take(s.len() as int) =~= s_spec);
86        }
87        sum
88    }

162    // Lemma: sum(take(i+1)) = sum(take(i)) + s[i]@ for CheckedU32 sequences
163    pub proof fn lemma_sum_checked_u32_unfold_take(s: Seq<CheckedU32>, i: int)
164        requires 0 <= i < s.len()
165        ensures spec_sum_checked_u32_seq(s.take(i + 1)) == spec_sum_checked_u32_seq(s.take(i)) + s[i]@
166    {
167        let views = s.map(|j: int, c: CheckedU32| c@);
168        let take_new_views = s.take(i + 1).map(|j: int, c: CheckedU32| c@);
169        let take_old_views = s.take(i).map(|j: int, c: CheckedU32| c@);
170        
171        // Show: s.take(i+1).map(f) == views.take(i+1)
172        assert(take_new_views =~= views.take(i + 1));
173        assert(take_old_views =~= views.take(i));
174        
175        // Now use the int lemma
176        lemma_sum_int_unfold_take(views, i);
177    }
